"seleccionar 'pais desde' y 'pais hasta' de la tabla
"SPFLI donde el CARRID= 'AA' y el CONNID = '0017'

DATA: lv_paisf TYPE LAND1,
      lv_paish TYPE LAND1.

SELECT SINGLE COUNTRYFR COUNTRYTO 
FROM splfi
INTO (lv_paisf,lv_paish)
WHERE carrid = 'AA' and connid = '0017'.

write: 'Los paises para carrid = AA and connid = 0017 son: ',
	lv_paisf, lv_paish.

"Seleccionar 'Hora salida', 'Hora llegada' y 'Distancia' de la
"tabla SPFLI donde el CARRID = 'XX' y el CONNID = 'XXXX'.

DATA: lv_timed TYPE S_DEP_TIME,
      lv_timea TYPE S_ARR_TIME,
      lv_distancia TYPE S_DISTANCE.

SELECT SINGLE DEPTIME ARRTIME DISTANCE
FROM spfli
INTO (lv_timed,lv_timea,lv_distancia)
WHERE carrid = 'AA' AND connid = '0017'.

IF sy-subrc = 0.
 SKIP.
 WRITE: 'Los datos para carrid = AA AND connid = 0017 son: ',
	lv_timed, lv_timea, lv_distancia.
ELSE.
 SKIP.
 WRITE: 'No hay datos para la selección'.
ENDIF.

"Seleccionar todos los registros de SPFLI para CARRID = AA.

DATA: it_spfli TYPE standard table of spfli,
      wa_spfli LIKE LINE OF it_spfli.

SELECT *
FROM spfli
INTO TABLE it_spfli
WHERE carrid = 'AA'.

IF sy-subrc = 0.
 SKIP.
 WRITE: 'El contenido seleccionado es: '.
 SKIP.
LOOP AT it_spfli INTO wa_spfli.	  "wa = fila de una tabla
 WRITE: wa_spfli-carrid, wa_spfli-connid, wa_spfli-cityfrom.
 SKIP.
ENDLOOP.

ELSE.
 SKIP.
 WRITE: 'No hay datos para la selección'.
ENDIF.
